Convertir XML a CSV

  1. # Importing the required libraries
  2. import xml.etree.ElementTree as Xet
  3. import pandas as pd
  4.  
  5. cols = ["device_count", "device_pool", "device_type"]
  6. rows = []
  7.  
  8. # Parsing the XML file
  9. xmlparse = Xet.parse('response.xml')
  10. root = xmlparse.getroot()
  11.  
  12. result = root.iterfind(".//row/*")
  13. for i in result:
  14.     escribe = False
  15.     if i.tag == "device_count":
  16.         device_count = i.text
  17.     elif i.tag == "device_pool":
  18.         device_pool = i.text
  19.     elif i.tag == "device_type":
  20.         device_type = i.text
  21.         escribe = True
  22.     if escribe:
  23.         rows.append({"device_count": device_count,
  24.                  "device_pool": device_pool,
  25.                  "device_type": device_type})
  26.  
  27. df = pd.DataFrame(rows, columns=cols)
  28.  
  29. # Writing dataframe to csv
  30. df.to_csv('output.csv')
