Convertir XML a CSV 

                    # Importing the required libraries
                    import xml.etree.ElementTree as Xet
                    import pandas as pd
                      
                    cols = ["device_count", "device_pool", "device_type"]
                    rows = []
                      
                    # Parsing the XML file
                    xmlparse = Xet.parse('response.xml')
                    root = xmlparse.getroot()

                    result = root.iterfind(".//row/*")
                    for i in result:
                        escribe = False
                        if i.tag == "device_count":
                            device_count = i.text
                        elif i.tag == "device_pool":
                            device_pool = i.text
                        elif i.tag == "device_type":
                            device_type = i.text
                            escribe = True
                        if escribe:
                            rows.append({"device_count": device_count,
                                     "device_pool": device_pool,
                                     "device_type": device_type})
                      
                    df = pd.DataFrame(rows, columns=cols)
                      
                    # Writing dataframe to csv
                    df.to_csv('output.csv')
